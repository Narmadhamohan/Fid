Step 1: Create Sample Tables
CREATE TABLE employees_A (
    employee_id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    salary NUMERIC(10, 2)
);

CREATE TABLE employees_B (
    employee_id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    salary NUMERIC(10, 2)
);
Step 2: Insert Sample Data

    ('John Doe', 60000),
    ('Alice Johnson', 70000),
    ('Sara Lee', 80000);

1. UNION
SELECT name, salary FROM employees_A
UNION
SELECT name, salary FROM employees_B;
2. 2. UNION ALL
SELECT name, salary FROM employees_A
UNION ALL
SELECT name, salary FROM employees_B;
3.INTERSECT
SELECT name, salary FROM employees_A
INTERSECT
SELECT name, salary FROM employees_B;

4. EXCEPT
SELECT name, salary FROM employees_A
EXCEPT
SELECT name, salary FROM employees_B;
5. 5. Comparison Operators: >, <, > ALL, < ANY, > ANY
SELECT name, salary
FROM employees_A
WHERE salary > 60000;
SELECT name, salary
FROM employees_B
WHERE salary < 70000;
SELECT name, salary
FROM employees_A
WHERE salary > ALL (SELECT salary FROM employees_B);
SELECT name, salary
FROM employees_B
WHERE salary < ANY (SELECT salary FROM employees_A);
SELECT name, salary
FROM employees_A
WHERE salary > ANY (SELECT salary FROM employees_B);
