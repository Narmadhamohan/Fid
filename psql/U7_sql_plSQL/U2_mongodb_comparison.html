SQL vs MongoDB Queries

SQL: Insert

INSERT INTO users (name, age) VALUES ('Alice', 25);

MongoDB: Insert

db.users.insertOne({ name: "Alice", age: 25 });

SQL: Select All

SELECT * FROM users;

MongoDB: Select All

db.users.find();

SQL: Select Specific Fields

SELECT name, age FROM users;

MongoDB: Select Specific Fields

db.users.find({}, { name: 1, age: 1, _id: 0 });

SQL: Where Clause

SELECT * FROM users WHERE age > 20;

MongoDB: Where Clause

db.users.find({ age: { $gt: 20 } });

SQL: Update

UPDATE users SET age = 26 WHERE name = 'Alice';

MongoDB: Update

db.users.updateOne({ name: "Alice" }, { $set: { age: 26 } });

SQL: Delete

DELETE FROM users WHERE name = 'Alice';

MongoDB: Delete

db.users.deleteOne({ name: "Alice" });

SQL: Order By

SELECT * FROM users ORDER BY age DESC;

MongoDB: Order By

db.users.find().sort({ age: -1 });

SQL: Limit

SELECT * FROM users LIMIT 5;

MongoDB: Limit

db.users.find().limit(5);

SQL: Count

SELECT COUNT(*) FROM users;

MongoDB: Count

db.users.countDocuments();

SQL: Join

SELECT users.name, orders.item FROM users
JOIN orders ON users.user_id = orders.user_id;

MongoDB: Join (Aggregation)

db.users.aggregate([
  { $lookup: {
      from: "orders",
      localField: "user_id",
      foreignField: "user_id",
      as: "orders_info"
  }},
  { $unwind: "$orders_info" },
  { $project: { name: 1, "orders_info.item": 1, _id: 0 } }
]);
