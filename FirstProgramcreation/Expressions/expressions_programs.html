Example Program 1:

class BlockDemo {
     public static void main(String[] args) {
          boolean condition = true;
          if (condition) { // begin block 1
               System.out.println("Condition is true.");
          } // end block one
          else { // begin block 2
               System.out.println("Condition is false.");
          } // end block 2
     }
}

Example Program 2:

// Python Example
def factorial(n):
    if n == 0:
        return 1
    return n * factorial(n-1)

print(factorial(5))
